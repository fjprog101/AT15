package org.fundacionjala.at15.katas.pokerhands.mauricio;

public class ClubCard extends Card {

    private String suit;

    public ClubCard(String value) {
        super(value);
        this.suit = "C";
    }

    public String getClubCard() {
        return getValue() + suit;
    }

}
